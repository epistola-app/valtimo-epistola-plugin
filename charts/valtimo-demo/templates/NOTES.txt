  Then open: http://localhost:8080

{{- end }}

  - The backend may take 60-90 seconds to fully start (Spring Boot + DB migrations).
  - Check pod status: kubectl get pods -l app.kubernetes.io/instance={{ .Release.Name }}
  - Check backend logs: kubectl logs -l app.kubernetes.io/component=backend -f
